			\Titem Aplicación móvil con funcionalidad OCR para escaneo de tickets
			\Titem Sistema de categorización automática de gastos
			\Titem Módulo de análisis predictivo y reportes
			\Titem Sistema de notificaciones y alertas
			\Titem Integración con APIs bancarias básicas
		\end{Titemize}
	}
	\ESCOMPchItem{Excluye:}{

			\Titem Integración con todas las instituciones bancarias
			\Titem Funcionalidades de inversión avanzada
			\Titem Sistema de pagos peer-to-peer
			\Titem Asesoría financiera personalizada en tiempo real
		\end{Titemize}
	}
