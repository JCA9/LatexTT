%=========================================================
% Interfaz de usuario: Dashboard principal

\begin{interfaceUsario}{IU-001}{Dashboard principal}{
	Pantalla principal que muestra un resumen completo del estado financiero del usuario, incluyendo balance actual, gastos recientes, progreso de presupuestos y acceso rápido a funciones principales.
}
	\IUactores{\cdtActorRef{Usuario Joven}{Usuario autenticado que accede a la información principal de sus finanzas}}
	\IUitem[Tipo]{Pantalla principal}
	\IUitem[Plataforma]{Móvil y Web responsive}
	\IUitem[Prioridad]{Alta}
	\IUitem[Estatus]{Activo}
	\IUitem[Fecha]{Noviembre 2024}

	% Elementos de la interfaz
	\IUelem{Header personalizado}{
		\IUli{Saludo personalizado con nombre del usuario}
		\IUli{Icono de notificaciones con badge de count}
		\IUli{Avatar del usuario con acceso a menú de perfil}
		\IUli{Fecha actual y último update de datos}
	}

	\IUelem{Resumen financiero}{
		\IUli{Card principal con balance actual destacado}
		\IUli{Indicador visual de tendencia (↑/↓) del balance}
		\IUli{Gastos del mes actual vs mes anterior}
		\IUli{Ingresos del mes con comparación histórica}
	}

	\IUelem{Gráfico de gastos}{
		\IUli{Gráfico circular (donut) de gastos por categoría del mes}
		\IUli{Colores distintivos para cada categoría}
		\IUli{Porcentajes y montos al hacer tap en cada segmento}
		\IUli{Opción de cambiar período (semana/mes/trimestre)}
	}

	\IUelem{Transacciones recientes}{
		\IUli{Lista de las últimas 5 transacciones}
		\IUli{Icono de categoría, descripción, monto y fecha}
		\IUli{Indicador visual para gastos (rojo) vs ingresos (verde)}
		\IUli{Botón ``Ver todas'' para acceder al historial completo}
	}

	\IUelem{Progreso de presupuestos}{
		\IUli{Cards de presupuestos activos con barras de progreso}
		\IUli{Código de colores: verde (dentro), amarillo (70\%), rojo (90\%+)}
		\IUli{Monto gastado / monto total para cada presupuesto}
		\IUli{Días restantes del período de cada presupuesto}
	}

	\IUelem{Acciones rápidas}{
		\IUli{Floating Action Button para ``Registrar gasto'' (móvil)}
		\IUli{Botones de acceso rápido: ``Escanear ticket'', ``Gasto manual'', ``Ingreso''}
		\IUli{Acceso directo a ``Analytics'' y ``Presupuestos''}
		\IUli{Widget de ``Consejos del día'' con educación financiera}
	}

	\IUelem{Navegación}{
		\IUli{Tab bar inferior (móvil): Inicio, Gastos, Analytics, Presupuestos, Perfil}
		\IUli{Sidebar lateral (web): Menú expandible con todas las secciones}
		\IUli{Breadcrumb de navegación (web) para orientación del usuario}
	}

	% Interacciones principales
	\IUinter{
		\IUli{Tap en balance: Expandir detalles de ingresos y gastos del período}
		\IUli{Tap en segmento del gráfico: Filtrar transacciones por esa categoría}
		\IUli{Swipe en transacción: Mostrar opciones ``Editar'' y ``Categorizar''}
		\IUli{Pull-to-refresh: Sincronizar datos más recientes}
		\IUli{Tap en presupuesto: Navegar a detalles del presupuesto específico}
		\IUli{Long press en FAB: Mostrar menú con opciones de registro rápido}
	}

	% Estados de la interfaz
	\IUestados{
		\IUli{\textbf{Carga inicial}: Skeleton screens mientras cargan los datos}
		\IUli{\textbf{Sincronizando}: Indicador discreto de sincronización en progreso}
		\IUli{\textbf{Sin datos}: Empty state para usuarios nuevos con onboarding}
		\IUli{\textbf{Error de conexión}: Mensaje con datos en cache y opción de reintento}
		\IUli{\textbf{Alerta de presupuesto}: Banner destacado cuando se excede un límite}
	}

	% Validaciones y reglas
	\IUval{
		\IUli{Los montos se muestran en formato mexicano (\$X,XXX.XX)}
		\IUli{Las fechas siguen formato dd/mm/yyyy}
		\IUli{Los colores de categorías son consistentes en toda la aplicación}
		\IUli{Los datos se actualizan automáticamente cada 30 segundos si hay conexión}
		\IUli{Se respeta el tema del sistema (claro/oscuro) del usuario}
	}

	% Responsive design
	\IUresp{
		\IUli{\textbf{Mobile (320-767px)}: Layout vertical, cards apiladas, FAB visible}
		\IUli{\textbf{Tablet (768-1023px)}: Layout de 2 columnas, sidebar colapsible}
		\IUli{\textbf{Desktop (1024px+)}: Layout de 3 columnas, sidebar fijo, widgets adicionales}
		\IUli{\textbf{Adaptaciones}: Tamaño de texto escalable, touch targets de 44px mínimo}
	}

	% Referencias
	\IUref{
		\UCref{CU-001}{Registrar gasto mediante OCR}
		\UCref{CU-003}{Consultar analytics y predicciones}
		\BRref{RN-011}{Sincronización de datos}
	}
\end{interfaceUsario}
