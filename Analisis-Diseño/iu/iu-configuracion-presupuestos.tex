% Interfaz de usuario: Configuración de presupuestos

\begin{interfaceUsario}{IU-003}{Configuración de presupuestos}{
	Interfaz completa para la creación, configuración y gestión de presupuestos personalizados, permitiendo al usuario establecer límites de gasto por categorías y períodos específicos con alertas personalizables.
}
	\IUactores{\cdtActorRef{Usuario Joven}{Usuario que desea establecer y gestionar presupuestos para controlar sus gastos}}
	\IUitem[Tipo]{Pantalla de configuración}
	\IUitem[Plataforma]{Móvil y Web}
	\IUitem[Prioridad]{Media}
	\IUitem[Estatus]{Activo}
	\IUitem[Fecha]{Noviembre 2024}

	% Elementos de la interfaz
	\IUelem{Header de presupuestos}{
		\IUli{Título ``Mis Presupuestos'' con contador de activos/total}
		\IUli{Botón ``+ Crear presupuesto'' prominente}
		\IUli{Toggle para mostrar presupuestos inactivos/archivados}
		\IUli{Filtros: Todos, Activos, Excedidos, Por vencer}
	}

	\IUelem{Lista de presupuestos existentes}{
		\IUli{Cards de presupuestos con información clave}
		\IUli{Nombre, período, progreso visual con barra coloreada}
		\IUli{Monto gastado / monto total con porcentaje}
		\IUli{Días restantes del período actual}
		\IUli{Estado visual: Normal (verde), Alerta (amarillo), Excedido (rojo)}
	}

	\IUelem{Formulario de creación/edición}{
		\IUli{Campo nombre del presupuesto con sugerencias inteligentes}
		\IUli{Input de monto con teclado numérico y formato automático}
		\IUli{Selector de período: Semanal, Mensual, Trimestral}
		\IUli{Date picker para fecha de inicio (por defecto: hoy)}
		\IUli{Multi-selector de categorías con iconos y colores}
	}

	\IUelem{Configuración avanzada}{
		\IUli{Slider para umbrales de alerta (70\%, 90\% por defecto)}
		\IUli{Checkboxes para tipos de notificación: Push, Email, In-app}
		\IUli{Toggle para presupuesto auto-renovable}
		\IUli{Selector de frecuencia de reportes: Diario, Semanal, Al final}
	}

	\IUelem{Analytics del presupuesto}{
		\IUli{Gráfico de progreso histórico de presupuestos similares}
		\IUli{Comparación con promedio de usuarios similares (anónimo)}
		\IUli{Predicción de cumplimiento basada en patrón actual}
		\IUli{Sugerencias de optimización de monto}
	}

	\IUelem{Vista detalle de presupuesto}{
		\IUli{Gráfico de progreso diario/semanal del período actual}
		\IUli{Lista de transacciones que contribuyen al presupuesto}
		\IUli{Distribución por subcategorías si aplica}
		\IUli{Historial de alertas generadas}
	}

	% Interacciones principales
	\IUinter{
		\IUli{Tap en card de presupuesto: Abrir vista detalle}
		\IUli{Swipe left en presupuesto: Mostrar opciones ``Editar'' y ``Pausar''}
		\IUli{Tap en barra de progreso: Mostrar desglose detallado}
		\IUli{Long press en categoría: Mostrar gastos históricos de esa categoría}
		\IUli{Pull to refresh: Actualizar datos y recalcular progreso}
		\IUli{Tap en ``Sugerencias'': Expandir recomendaciones de IA}
	}

	% Estados de la interfaz
	\IUestados{
		\IUli{\textbf{Lista vacía}: Empty state con ilustración y call-to-action}
		\IUli{\textbf{Creando presupuesto}: Formulario paso a paso con progreso}
		\IUli{\textbf{Calculando sugerencias}: Loading state para recomendaciones de IA}
		\IUli{\textbf{Presupuesto excedido}: Estado de alerta con opciones de acción}
		\IUli{\textbf{Sincronizando}: Indicador discreto durante actualización de datos}
	}

	% Validaciones y reglas
	\IUval{
		\IUli{El monto del presupuesto debe ser mayor a \$0 y menor a \$100,000}
		\IUli{La fecha de inicio no puede ser anterior a 30 días}
		\IUli{Debe seleccionarse al menos una categoría}
		\IUli{El nombre debe ser único entre presupuestos activos del usuario}
		\IUli{No se pueden crear más de 10 presupuestos activos simultáneamente}
	}

	% Sugerencias inteligentes
	\IUsmart{
		\IUli{\textbf{Monto sugerido}: Basado en gasto promedio histórico + 10\% buffer}
		\IUli{\textbf{Categorías frecuentes}: Pre-seleccionar categorías con más gastos}
		\IUli{\textbf{Período óptimo}: Sugerir período basado en patrones de ingreso}
		\IUli{\textbf{Alertas personalizadas}: Umbrales basados en comportamiento histórico}
		\IUli{\textbf{Auto-ajuste}: Sugerir modificaciones cuando se excede consistentemente}
	}

	% Gamificación
	\IUgamification{
		\IUli{Badges por cumplir presupuestos consecutivos}
		\IUli{Streak counter de presupuestos exitosos}
		\IUli{Comparación anónima con amigos (si optaron por ello)}
		\IUli{Celebración visual cuando se cumple un presupuesto}
		\IUli{Progress ring que se llena conforme avanza el período}
	}

	% Responsive adaptations
	\IUresp{
		\IUli{\textbf{Mobile}: Lista vertical, formulario en pantalla completa}
		\IUli{\textbf{Tablet}: Grid de 2 columnas, formulario en modal}
		\IUli{\textbf{Desktop}: Grid de 3 columnas, panel lateral para edición}
		\IUli{\textbf{Todas}: Gráficos adaptativos, texto escalable}
	}

	% Referencias
	\IUref{
		\UCref{CU-002}{Configurar presupuesto personal}
		\BRref{RN-007}{Presupuestos activos}
		\BRref{RN-008}{Período de presupuesto}
		\BRref{RN-009}{Alertas de presupuesto}
		\MSGref{MSG-005}{Presupuesto 70\%}
		\MSGref{MSG-006}{Presupuesto excedido}
	}
\end{interfaceUsario}
